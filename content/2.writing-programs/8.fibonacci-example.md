## Guest: `guest/src/main.rs`

```rust
#![no_main]
ziskos::entrypoint!(main);

use ziskos::io;
use serde::{Serialize, Deserialize};

#[derive(Serialize, Deserialize, Debug)]
struct FibResult {
    n: u32,
    value: u64,
}

fn main() {
    let n: u32 = io::read();
    println!("Guest received n = {}", n);

    let mut a: u64 = 0;
    let mut b: u64 = 1;
    for _ in 0..n {
        let temp = b;
        b = a + b;
        a = temp;
    }

    let result = FibResult { n, value: a };
    println!("Fibonacci({}) = {}", n, a);
    io::commit(&result);
}
```

The guest reads `n` from the host, computes the nth Fibonacci number, and commits the result as public output. See [Project Setup](/writing-programs/project-setup) for a detailed walkthrough of this code.

## Build script: `host/build.rs`

```rust
use std::path::PathBuf;
use zisk_sdk::{ZiskIO, ZiskStdin, build_program};

fn main() {
    build_program("../guest");

    let n = 10u32;
    let stdin = ZiskStdin::new();
    stdin.write(&n);

    let path = PathBuf::from("tmp/input.bin");
    if let Some(parent) = path.parent() {
        std::fs::create_dir_all(parent).unwrap();
    }
    stdin.save(&path).unwrap();
}
```

Runs at `cargo build` time. Cross-compiles the guest to a RISC-V ELF and writes a serialized input file with `n = 10`. See [Build and Run](/writing-programs/build-and-run) for details.

## Host: `host/src/main.rs`

This is the primary host program. It executes the guest, generates a VADCOP STARK proof, and verifies it:

```rust
use anyhow::Result;
use serde::{Deserialize, Serialize};
use zisk_sdk::{
    ZiskStdin, ZiskIO, ElfBinary,
    ProofOpts, ProverClient, include_elf,
};

pub const ELF: ElfBinary = include_elf!("guest");

#[derive(Serialize, Deserialize, Debug)]
struct FibResult {
    n: u32,
    value: u64,
}

fn main() -> Result<()> {
    let n = 10u32;
    let stdin = ZiskStdin::new();
    stdin.write(&n);

    let client = ProverClient::builder().emu().build().unwrap();
    let (pk, _) = client.setup(&ELF)?;

    // Execute first to verify the program works
    let result = client.execute(&pk, stdin)?;
    println!("Cycles: {}", result.get_execution_steps());

    let output: FibResult = result.get_public_values()?;
    println!("Fibonacci({}) = {}", output.n, output.value);

    // Generate a proof
    let stdin = ZiskStdin::new();
    stdin.write(&n);

    let proof_opts = ProofOpts::default().minimal_memory();
    let result = client.prove(&pk, stdin)
        .with_proof_options(proof_opts)
        .run()?;

    println!("Execution steps: {}", result.get_execution_steps());

    client.verify(result.get_proof(), result.get_publics(), result.get_program_vk())?;
    println!("Proof verified!");

    let fib: FibResult = result.get_public_values()?;
    println!("Proven result: Fibonacci({}) = {}", fib.n, fib.value);

    Ok(())
}
```

Run from the project root:

```bash
cargo run --release
```
